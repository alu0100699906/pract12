def % (object)

    division=self.abs/object.abs

def gcd(n, d)
    n1,n2 = n.abs,d.abs
    gcd = 1
    k = 1
    while (k <= n1 and k <= n2)
         if (n1 % k == 0 and n2 % k == 0)
             gcd = k
         end
         k += 1
    end
    return gcd
end

def mcm(a,b)
   #mediante el metodo de euclides, resolvemos el minimo comun multiplo mediante el
   #maximo comun divisor
   aux=gcd(a,b)
   (a/aux)*b
end
